\subsection{Orthogonality of Vectors and Subspaces}
\begin{itemize}
    \item Orthogonal vectors have $\mathbf{v}^T \mathbf{w} = 0$, so $\|\mathbf{v}\|^2 +
              \|\mathbf{w}\|^2 = \|\mathbf{v} + \mathbf{w}\|^2$
    \item Subspaces $V$ and $W$ are ortogonal when $\mathbf{v}^T \mathbf{w} = 0$ for
          every $\mathbf{v}$ in $V$ and $\mathbf{w}$ in $W$
    \item The row space $\mathbf{C}(A^T)$ is orthogonal to the nullspace $\mathbf{N}(A)$.
          The column space $\mathbf{C}(A)$ is orthogonal to left nullspace
          $\mathbf{N}(A^T)$.
    \item The dimensions of $\mathbf{C}(A^T)$ and $\mathbf{N}(A)$ add to $r+(n-r)=n$. The
          dmensions of $\mathbf{C}(A)$ and $\mathbf{N}(A^T)$ add to $r+(m-r)=m$:
          Orthogonal Complements.
    \item If $n$ vectors in $\mathbb{R}^n$ are independent, the span $\mathbb{R}^n$. If
          $n$ vectors span $\mathbb{R}^n$, the are independent.
    \item if $V$ and $W$ are orthogonal subspaces in $\mathbb{R}^n$ then $dim(V) + dim(W)
              \le n$
\end{itemize}
